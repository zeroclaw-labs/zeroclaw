1//! Shell-based tool derived from a skill's `[[tools]]` section.
2//!
3//! Each `SkillTool` with `kind = "shell"` or `kind = "script"` is converted
4//! into a `SkillShellTool` that implements the `Tool` trait. The tool name is
5//! prefixed with the skill name (e.g. `my_skill.run_lint`) to avoid collisions
6//! with built-in tools.
7
8use crate::security::SecurityPolicy;
9use async_trait::async_trait;
10use std::collections::HashMap;
11use std::sync::Arc;
12use std::time::Duration;
13use zeroclaw_api::tool::{Tool, ToolResult};
14
15/// Maximum execution time for a skill shell command (seconds).
16const SKILL_SHELL_TIMEOUT_SECS: u64 = 60;
17/// Maximum output size in bytes (1 MB).
18const MAX_OUTPUT_BYTES: usize = 1_048_576;
19
20/// A tool derived from a skill's `[[tools]]` section that executes shell commands.
21pub struct SkillShellTool {
22    tool_name: String,
23    tool_description: String,
24    command_template: String,
25    args: HashMap<String, String>,
26    security: Arc<SecurityPolicy>,
27}
28
29impl SkillShellTool {
30    /// Create a new skill shell tool.
31    ///
32    /// The tool name is prefixed with the skill name (`skill_name.tool_name`)
33    /// to prevent collisions with built-in tools.
34    pub fn new(
35        skill_name: &str,
36        tool: &crate::skills::SkillTool,
37        security: Arc<SecurityPolicy>,
38    ) -> Self {
39        Self {
40            tool_name: format!("{}.{}", skill_name, tool.name),
41            tool_description: tool.description.clone(),
42            command_template: tool.command.clone(),
43            args: tool.args.clone(),
44            security,
45        }
46    }
47
48    fn build_parameters_schema(&self) -> serde_json::Value {
49        let mut properties = serde_json::Map::new();
50        let mut required = Vec::new();
51
52        for (name, description) in &self.args {
53            properties.insert(
54                name.clone(),
55                serde_json::json!({
56                    "type": "string",
57                    "description": description
58                }),
59            );
60            required.push(serde_json::Value::String(name.clone()));
61        }
62
63        serde_json::json!({
64            "type": "object",
65            "properties": properties,
66            "required": required
67        })
68    }
69
70    /// Substitute `{{arg_name}}` placeholders in the command template with
71    /// the provided argument values. Unknown placeholders are left as-is.
72    fn substitute_args(&self, args: &serde_json::Value) -> String {
73        let mut command = self.command_template.clone();
74        if let Some(obj) = args.as_object() {
75            for (key, value) in obj {
76                let placeholder = format!("{{{{{}}}}}", key);
77                let replacement = value.as_str().unwrap_or_default();
78                command = command.replace(&placeholder, replacement);
79            }
80        }
81        command
82    }
83}
84
85#[async_trait]
86impl Tool for SkillShellTool {
87    fn name(&self) -> &str {
88        &self.tool_name
89    }
90
91    fn description(&self) -> &str {
92        &self.tool_description
93    }
94
95    fn parameters_schema(&self) -> serde_json::Value {
96        self.build_parameters_schema()
97    }
98
99    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
100        let command = self.substitute_args(&args);
101
102        // Rate limit check
103        if self.security.is_rate_limited() {
104            return Ok(ToolResult {
105                success: false,
106                output: String::new(),
107                error: Some("Rate limit exceeded: too many actions in the last hour".into()),
108            });
109        }
110
111        // Security validation — always requires explicit approval (approved=true)
112        // since skill tools are user-defined and should be treated as medium-risk.
113        match self.security.validate_command_execution(&command, true) {
114            Ok(_) => {}
115            Err(reason) => {
116                return Ok(ToolResult {
117                    success: false,
118                    output: String::new(),
119                    error: Some(reason),
120                });
121            }
122        }
123
124        if let Some(path) = self.security.forbidden_path_argument(&command) {
125            return Ok(ToolResult {
126                success: false,
127                output: String::new(),
128                error: Some(format!("Path blocked by security policy: {path}")),
129            });
130        }
131
132        if !self.security.record_action() {
133            return Ok(ToolResult {
134                success: false,
135                output: String::new(),
136                error: Some("Rate limit exceeded: action budget exhausted".into()),
137            });
138        }
139
140        // Build and execute the command
141        let mut cmd = tokio::process::Command::new("sh");
142        cmd.arg("-c").arg(&command);
143        cmd.current_dir(&self.security.workspace_dir);
144        cmd.env_clear();
145
146        // Only pass safe environment variables
147        for var in &[
148            "PATH", "HOME", "TERM", "LANG", "LC_ALL", "USER", "SHELL", "TMPDIR",
149        ] {
150            if let Ok(val) = std::env::var(var) {
151                cmd.env(var, val);
152            }
153        }
154
155        let result =
156            tokio::time::timeout(Duration::from_secs(SKILL_SHELL_TIMEOUT_SECS), cmd.output()).await;
157
158        match result {
159            Ok(Ok(output)) => {
160                let mut stdout = String::from_utf8_lossy(&output.stdout).to_string();
161                let mut stderr = String::from_utf8_lossy(&output.stderr).to_string();
162
163                if stdout.len() > MAX_OUTPUT_BYTES {
164                    let mut b = MAX_OUTPUT_BYTES.min(stdout.len());
165                    while b > 0 && !stdout.is_char_boundary(b) {
166                        b -= 1;
167                    }
168                    stdout.truncate(b);
169                    stdout.push_str("\n... [output truncated at 1MB]");
170                }
171                if stderr.len() > MAX_OUTPUT_BYTES {
172                    let mut b = MAX_OUTPUT_BYTES.min(stderr.len());
173                    while b > 0 && !stderr.is_char_boundary(b) {
174                        b -= 1;
175                    }
176                    stderr.truncate(b);
177                    stderr.push_str("\n... [stderr truncated at 1MB]");
178                }
179
180                Ok(ToolResult {
181                    success: output.status.success(),
182                    output: stdout,
183                    error: if stderr.is_empty() {
184                        None
185                    } else {
186                        Some(stderr)
187                    },
188                })
189            }
190            Ok(Err(e)) => Ok(ToolResult {
191                success: false,
192                output: String::new(),
193                error: Some(format!("Failed to execute command: {e}")),
194            }),
195            Err(_) => Ok(ToolResult {
196                success: false,
197                output: String::new(),
198                error: Some(format!(
199                    "Command timed out after {SKILL_SHELL_TIMEOUT_SECS}s and was killed"
200                )),
201            }),
202        }
203    }
204}

206#[cfg(test)]
207mod tests {
208    use super::*;
209    use crate::security::{AutonomyLevel, SecurityPolicy};
210    use crate::skills::SkillTool;
211
212    fn test_security() -> Arc<SecurityPolicy> {
213        Arc::new(SecurityPolicy {
214            autonomy: AutonomyLevel::Full,
215            workspace_dir: std::env::temp_dir(),
216            ..SecurityPolicy::default()
217        })
218    }
219
220    fn sample_skill_tool() -> SkillTool {
221        let mut args = HashMap::new();
222        args.insert("file".to_string(), "The file to lint".to_string());
223        args.insert(
224            "format".to_string(),
225            "Output format (json|text)".to_string(),
226        );
227
228        SkillTool {
229            name: "run_lint".to_string(),
230            description: "Run the linter on a file".to_string(),
231            kind: "shell".to_string(),
232            command: "lint --file {{file}} --format {{format}}".to_string(),
233            args,
234        }
235    }
236
237    #[test]
238    fn skill_shell_tool_name_is_prefixed() {
239        let tool = SkillShellTool::new("my_skill", &sample_skill_tool(), test_security());
240        assert_eq!(tool.name(), "my_skill.run_lint");
241    }
242
243    #[test]
244    fn skill_shell_tool_description() {
245        let tool = SkillShellTool::new("my_skill", &sample_skill_tool(), test_security());
246        assert_eq!(tool.description(), "Run the linter on a file");
247    }
248
249    #[test]
250    fn skill_shell_tool_parameters_schema() {
251        let tool = SkillShellTool::new("my_skill", &sample_skill_tool(), test_security());
252        let schema = tool.parameters_schema();
253
254        assert_eq!(schema["type"], "object");
255        assert!(schema["properties"]["file"].is_object());
256        assert_eq!(schema["properties"]["file"]["type"], "string");
257        assert!(schema["properties"]["format"].is_object());
258
259        let required = schema["required"]
260            .as_array()
261            .expect("required should be array");
262        assert_eq!(required.len(), 2);
263    }
264
265    #[test]
266    fn skill_shell_tool_substitute_args() {
267        let tool = SkillShellTool::new("my_skill", &sample_skill_tool(), test_security());
268        let result = tool.substitute_args(&serde_json::json!({
269            "file": "src/main.rs",
270            "format": "json"
271        }));
272        assert_eq!(result, "lint --file src/main.rs --format json");
273    }
274
275    #[test]
276    fn skill_shell_tool_substitute_missing_arg() {
277        let tool = SkillShellTool::new("my_skill", &sample_skill_tool(), test_security());
278        let result = tool.substitute_args(&serde_json::json!({"file": "test.rs"}));
279        // Missing {{format}} placeholder stays in the command
280        assert!(result.contains("{{format}}"));
281        assert!(result.contains("test.rs"));
282    }

300    #[tokio::test]
301    async fn skill_shell_tool_executes_echo() {
302        let st = SkillTool {
303            name: "hello".to_string(),
304            description: "Say hello".to_string(),
305            kind: "shell".to_string(),
306            command: "echo hello-skill".to_string(),

310        let result = tool.execute(serde_json::json!({})).await.unwrap();
