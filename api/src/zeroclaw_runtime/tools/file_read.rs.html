9/// Read file contents with workspace sandboxing.

12}

14impl FileReadTool {
15    pub fn new(security: Arc<SecurityPolicy>) -> Self {
16        Self { security }
17    }
18
19    /// Validate and resolve a caller-supplied path to an absolute candidate.
20    /// Mirrors the logic in `FileWriteTool::resolve_candidate`.
21    fn resolve_candidate(&self, path: &str) -> anyhow::Result<std::path::PathBuf> {
22        let workspace_dir = &self.security.workspace_dir;
23
24        if path.contains('\0') {
25            anyhow::bail!("Path not allowed: contains null byte");
26        }
27        if std::path::Path::new(path)
28            .components()
29            .any(|c| matches!(c, std::path::Component::ParentDir))
30        {
31            anyhow::bail!("Path not allowed by security policy: {path}");
32        }
33
34        let p = std::path::Path::new(path);
35        if p.is_absolute() {
36            let workspace_canonical = workspace_dir
37                .canonicalize()
38                .unwrap_or_else(|_| workspace_dir.clone());
39            if p.starts_with(&workspace_canonical) || p.starts_with(workspace_dir.as_path()) {
40                return Ok(p.to_path_buf());
41            }
42            for root in &self.security.allowed_roots {
43                let root_canonical = root.canonicalize().unwrap_or_else(|_| root.clone());
44                if p.starts_with(&root_canonical) || p.starts_with(root.as_path()) {
45                    return Ok(p.to_path_buf());
46                }
47            }
48            anyhow::bail!("Path not allowed by security policy: {path}");
49        }
50
51        if let Ok(workspace_rootless) = workspace_dir.strip_prefix("/")
52            && let Ok(stripped) = p.strip_prefix(workspace_rootless)
53        {
54            return Ok(if stripped.as_os_str().is_empty() {
55                workspace_dir.clone()
56            } else {
57                workspace_dir.join(stripped)
58            });
59        }
60
61        Ok(workspace_dir.join(p))
62    }
63}

65#[async_trait]
66impl Tool for FileReadTool {
67    fn name(&self) -> &str {
68        "file_read"
69    }
70
71    fn description(&self) -> &str {
72        "Read file contents with line numbers. Supports partial reading via offset and limit. Extracts text from PDF; other binary files are read with lossy UTF-8 conversion."
73    }
74
75    fn parameters_schema(&self) -> serde_json::Value {
76        json!({
77            "type": "object",
78            "properties": {
79                "path": {
80                    "type": "string",
81                    "description": "Path to the file. Relative paths resolve from workspace root; absolute paths must be within the workspace."
82                },
83                "offset": {
84                    "type": "integer",
85                    "description": "Starting line number (1-based, default: 1)"
86                },
87                "limit": {
88                    "type": "integer",
89                    "description": "Maximum number of lines to return (default: all)"
90                }
91            },
92            "required": ["path"]
93        })
94    }
95
96    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<ToolResult> {
97        let path = args
98            .get("path")
99            .and_then(|v| v.as_str())
100            .ok_or_else(|| anyhow::anyhow!("Missing 'path' parameter"))?;
101
102        if self.security.is_rate_limited() {
103            return Ok(ToolResult {
104                success: false,
105                output: String::new(),
106                error: Some("Rate limit exceeded: too many actions in the last hour".into()),
107            });
108        }
109
110        // Validate and build candidate path using workspace_dir directly.
111        let full_path = match self.resolve_candidate(path) {
112            Ok(p) => p,
113            Err(e) => {
114                return Ok(ToolResult {
115                    success: false,
116                    output: String::new(),
117                    error: Some(e.to_string()),
118                });
119            }
120        };
121
122        // Record action BEFORE canonicalization so that every non-trivially-rejected
123        // request consumes rate limit budget. This prevents attackers from probing
124        // path existence (via canonicalize errors) without rate limit cost.
125        if !self.security.record_action() {
126            return Ok(ToolResult {
127                success: false,
128                output: String::new(),
129                error: Some("Rate limit exceeded: action budget exhausted".into()),
130            });
131        }
132
133        // Canonicalize to resolve symlinks, then enforce workspace boundary.
134        let resolved_path = match tokio::fs::canonicalize(&full_path).await {
135            Ok(p) => p,
136            Err(e) => {
137                return Ok(ToolResult {
138                    success: false,
139                    output: String::new(),
140                    error: Some(format!("Failed to resolve file path: {e}")),
141                });
142            }
143        };
144
145        let workspace_canonical = self
146            .security
147            .workspace_dir
148            .canonicalize()
149            .unwrap_or_else(|_| self.security.workspace_dir.clone());
150
151        let in_workspace = resolved_path.starts_with(&workspace_canonical);
152        let in_allowed_root = !in_workspace
153            && self.security.allowed_roots.iter().any(|root| {
154                let rc = root.canonicalize().unwrap_or_else(|_| root.clone());
155                resolved_path.starts_with(&rc)
156            });
157
158        if !in_workspace && !in_allowed_root {
159            return Ok(ToolResult {
160                success: false,
161                output: String::new(),
162                error: Some(format!("Path escapes workspace directory: {path}")),
163            });
164        }
165
166        // Check file size AFTER canonicalization to prevent TOCTOU symlink bypass
167        match tokio::fs::metadata(&resolved_path).await {
168            Ok(meta) => {
169                if meta.len() > MAX_FILE_SIZE_BYTES {
170                    return Ok(ToolResult {
171                        success: false,
172                        output: String::new(),
173                        error: Some(format!(
174                            "File too large: {} bytes (limit: {MAX_FILE_SIZE_BYTES} bytes)",
175                            meta.len()
176                        )),
177                    });
178                }
179            }
180            Err(e) => {
181                return Ok(ToolResult {
182                    success: false,
183                    output: String::new(),
184                    error: Some(format!("Failed to read file metadata: {e}")),
185                });
186            }
187        }
188
189        match tokio::fs::read_to_string(&resolved_path).await {
190            Ok(contents) => {
191                let lines: Vec<&str> = contents.lines().collect();
192                let total = lines.len();
193
194                if total == 0 {
195                    return Ok(ToolResult {
196                        success: true,
197                        output: String::new(),
198                        error: None,
199                    });
200                }
201
202                let offset = args
203                    .get("offset")
204                    .and_then(|v| v.as_u64())
205                    .map(|v| {
206                        usize::try_from(v.max(1))
207                            .unwrap_or(usize::MAX)
208                            .saturating_sub(1)
209                    })
210                    .unwrap_or(0);
211                let start = offset.min(total);
212
213                let end = match args.get("limit").and_then(|v| v.as_u64()) {
214                    Some(l) => {
215                        let limit = usize::try_from(l).unwrap_or(usize::MAX);
216                        (start.saturating_add(limit)).min(total)
217                    }
218                    None => total,
219                };
220
221                if start >= end {
222                    return Ok(ToolResult {
223                        success: true,
224                        output: format!("[No lines in range, file has {total} lines]"),
225                        error: None,
226                    });
227                }
228
229                let numbered: String = lines[start..end]
230                    .iter()
231                    .enumerate()
232                    .map(|(i, line)| format!("{}: {}", start + i + 1, line))
233                    .collect::<Vec<_>>()
234                    .join("\n");
235
236                let partial = start > 0 || end < total;
237                let summary = if partial {
238                    format!("\n[Lines {}-{} of {total}]", start + 1, end)
239                } else {
240                    format!("\n[{total} lines total]")
241                };
242
243                Ok(ToolResult {
244                    success: true,
245                    output: format!("{numbered}{summary}"),
246                    error: None,
247                })
248            }
249            Err(_) => {
250                // Not valid UTF-8 — read raw bytes and try to extract text
251                let bytes = tokio::fs::read(&resolved_path)
252                    .await
253                    .map_err(|e| anyhow::anyhow!("Failed to read file: {e}"))?;
254
255                if let Some(text) = try_extract_pdf_text(&bytes) {
256                    return Ok(ToolResult {
257                        success: true,
258                        output: text,
259                        error: None,
260                    });
261                }
262
263                // Lossy fallback — replaces invalid bytes with U+FFFD
264                let lossy = String::from_utf8_lossy(&bytes).into_owned();
265                Ok(ToolResult {
266                    success: true,
267                    output: lossy,
268                    error: None,
269                })
270            }
271        }
272    }
273}

275#[cfg(feature = "rag-pdf")]
276fn try_extract_pdf_text(bytes: &[u8]) -> Option<String> {
277    if bytes.len() < 5 || &bytes[..5] != b"%PDF-" {
278        return None;
279    }
280    let text = pdf_extract::extract_text_from_mem(bytes).ok()?;
281    if text.trim().is_empty() {
282        return None;
283    }
284    Some(text)
285}

287#[cfg(not(feature = "rag-pdf"))]
288fn try_extract_pdf_text(_bytes: &[u8]) -> Option<String> {
289    None
290}

753    mod e2e_helpers {
754        use crate::observability::{NoopObserver, Observer};
755        use std::sync::{Arc, Mutex};
756        use zeroclaw_config::schema::MemoryConfig;
757        use zeroclaw_memory::{self, Memory};
758        use zeroclaw_providers::{ChatMessage, ChatRequest, ChatResponse, Provider};
759
760        pub type SharedRequests = Arc<Mutex<Vec<Vec<ChatMessage>>>>;
761
762        pub struct RecordingProvider {
763            responses: Mutex<Vec<ChatResponse>>,
764            pub requests: SharedRequests,
765        }
766
767        impl RecordingProvider {
768            pub fn new(responses: Vec<ChatResponse>) -> (Self, SharedRequests) {
769                let requests: SharedRequests = Arc::new(Mutex::new(Vec::new()));
770                let provider = Self {
771                    responses: Mutex::new(responses),
772                    requests: requests.clone(),
773                };
774                (provider, requests)
775            }
776        }
777
778        #[async_trait::async_trait]
779        impl Provider for RecordingProvider {
780            async fn chat_with_system(
781                &self,
782                _system_prompt: Option<&str>,
783                _message: &str,
784                _model: &str,
785                _temperature: Option<f64>,
786            ) -> anyhow::Result<String> {
787                Ok("fallback".into())
788            }
789
790            async fn chat(
791                &self,
792                request: ChatRequest<'_>,
793                _model: &str,
794                _temperature: Option<f64>,
795            ) -> anyhow::Result<ChatResponse> {
796                self.requests
797                    .lock()
798                    .unwrap()
799                    .push(request.messages.to_vec());
800
801                let mut guard = self.responses.lock().unwrap();
802                if guard.is_empty() {
803                    return Ok(ChatResponse {
804                        text: Some("done".into()),
805                        tool_calls: vec![],
806                        usage: None,
807                        reasoning_content: None,
808                    });
809                }
810                Ok(guard.remove(0))
811            }
812        }
813
814        pub fn make_memory() -> Arc<dyn Memory> {
815            let cfg = MemoryConfig {
816                backend: "none".into(),
817                ..MemoryConfig::default()
818            };
819            Arc::from(zeroclaw_memory::create_memory(&cfg, &std::env::temp_dir(), None).unwrap())
820        }
821
822        pub fn make_observer() -> Arc<dyn Observer> {
823            Arc::from(NoopObserver {})
824        }
825    }
