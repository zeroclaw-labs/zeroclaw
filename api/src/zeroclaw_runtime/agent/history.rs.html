12/// Find the largest byte index `<= i` that is a valid char boundary.
13/// MSRV-compatible replacement for `str::floor_char_boundary` (stable in 1.91).
14pub fn floor_char_boundary(s: &str, i: usize) -> usize {
15    if i >= s.len() {
16        return s.len();
17    }
18    let mut pos = i;
19    while pos > 0 && !s.is_char_boundary(pos) {
20        pos -= 1;
21    }
22    pos
23}

25/// Truncate a tool result to `max_chars`, keeping head (2/3) + tail (1/3)
26/// with a marker in the middle. Returns input unchanged if within limit or
27/// `max_chars == 0` (disabled).
28pub fn truncate_tool_result(output: &str, max_chars: usize) -> String {
29    if max_chars == 0 || output.len() <= max_chars {
30        return output.to_string();
31    }
32    let head_len = max_chars * 2 / 3;
33    let tail_len = max_chars.saturating_sub(head_len);
34    let head_end = floor_char_boundary(output, head_len);
35    // ceil_char_boundary: find smallest byte index >= i on a char boundary
36    let tail_start_raw = output.len().saturating_sub(tail_len);
37    let tail_start = if tail_start_raw >= output.len() {
38        output.len()
39    } else {
40        let mut pos = tail_start_raw;
41        while pos < output.len() && !output.is_char_boundary(pos) {
42            pos += 1;
43        }
44        pos
45    };
46    // Guard against overlap when max_chars is very small
47    if head_end >= tail_start {
48        return output[..floor_char_boundary(output, max_chars)].to_string();
49    }
50    let truncated_chars = tail_start - head_end;
51    format!(
52        "{}\n\n[... {} characters truncated ...]\n\n{}",
53        &output[..head_end],
54        truncated_chars,
55        &output[tail_start..]
56    )
57}

59/// Truncate a tool message's content, preserving JSON structure when the
60/// message stores `tool_call_id` alongside `content` (native tool-call
61/// format). Without this, `truncate_tool_result` destroys the JSON envelope
62/// and downstream providers receive a `null` `call_id` (#5425).
63pub fn truncate_tool_message(msg_content: &str, max_chars: usize) -> String {
64    if max_chars == 0 || msg_content.len() <= max_chars {
65        return msg_content.to_string();
66    }
67    if let Ok(mut obj) =
68        serde_json::from_str::<serde_json::Map<String, serde_json::Value>>(msg_content)
69        && obj.contains_key("tool_call_id")
70        && let Some(serde_json::Value::String(inner)) = obj.get("content")
71    {
72        let truncated = truncate_tool_result(inner, max_chars);
73        obj.insert("content".to_string(), serde_json::Value::String(truncated));
74        return serde_json::to_string(&obj).unwrap_or_else(|_| msg_content.to_string());
75    }
76    truncate_tool_result(msg_content, max_chars)
77}

79/// Aggressively trim old tool result messages in history to recover from
80/// context overflow. Keeps the last `protect_last_n` messages untouched.
81/// Returns total characters saved.
82pub fn fast_trim_tool_results(
83    history: &mut [zeroclaw_providers::ChatMessage],
84    protect_last_n: usize,
85) -> usize {
86    let trim_to = 2000;
87    let mut saved = 0;
88    let cutoff = history.len().saturating_sub(protect_last_n);
89    for msg in &mut history[..cutoff] {
90        if msg.role == "tool" && msg.content.len() > trim_to {
91            let original_len = msg.content.len();
92            msg.content = truncate_tool_message(&msg.content, trim_to);
93            saved += original_len - msg.content.len();
94        }
95    }
96    saved
97}

99/// Emergency: drop oldest non-system, non-recent messages from history.
100/// Tool groups (assistant + consecutive tool messages) are dropped
101/// atomically to preserve tool_use/tool_result pairing. See #4810.
102/// Returns number of messages dropped.
103pub fn emergency_history_trim(
104    history: &mut Vec<zeroclaw_providers::ChatMessage>,
105    keep_recent: usize,
106) -> usize {
107    let mut dropped = 0;
108    let target_drop = history.len() / 3;
109    let mut i = 0;
110    while dropped < target_drop && i < history.len().saturating_sub(keep_recent) {
111        if history[i].role == "system" {
112            i += 1;
113        } else if history[i].role == "assistant" {
114            // Count following tool messages — drop as atomic group
115            let mut tool_count = 0;
116            while i + 1 + tool_count < history.len().saturating_sub(keep_recent)
117                && history[i + 1 + tool_count].role == "tool"
118            {
119                tool_count += 1;
120            }
121            for _ in 0..=tool_count {
122                history.remove(i);
123                dropped += 1;
124            }
125        } else {
126            history.remove(i);
127            dropped += 1;
128        }
129    }
130    dropped += remove_orphaned_tool_messages(history);
131    dropped
132}

134/// Estimate token count for a message history using ~4 chars/token heuristic.
135/// Includes a small overhead per message for role/framing tokens.
136pub fn estimate_history_tokens(history: &[ChatMessage]) -> usize {
137    history
138        .iter()
139        .map(|m| {
140            // ~4 chars per token + ~4 framing tokens per message (role, delimiters)
141            m.content.len().div_ceil(4) + 4
142        })
143        .sum()
144}

146/// Trim conversation history to prevent unbounded growth.
147/// Preserves the system prompt (first message if role=system) and the most recent messages.
148pub fn trim_history(history: &mut Vec<ChatMessage>, max_history: usize) {
149    // Nothing to trim if within limit
150    let has_system = history.first().is_some_and(|m| m.role == "system");
151    let non_system_count = if has_system {
152        history.len() - 1
153    } else {
154        history.len()
155    };
156
157    if non_system_count <= max_history {
158        return;
159    }
160
161    let start = if has_system { 1 } else { 0 };
162    let to_remove = non_system_count - max_history;
163    history.drain(start..start + to_remove);
164    remove_orphaned_tool_messages(history);
165}

182pub fn load_interactive_session_history(
183    path: &Path,
184    system_prompt: &str,
185) -> Result<Vec<ChatMessage>> {
186    if !path.exists() {
187        return Ok(vec![ChatMessage::system(system_prompt)]);
188    }
189
190    let raw = std::fs::read_to_string(path)?;
191    let mut state: InteractiveSessionState = serde_json::from_str(&raw)?;
192    if state.history.is_empty() {
193        state.history.push(ChatMessage::system(system_prompt));
194    } else if state.history.first().map(|msg| msg.role.as_str()) != Some("system") {
195        state.history.insert(0, ChatMessage::system(system_prompt));
196    }
197
198    // Self-heal persisted sessions that were written with orphaned
199    // tool_result messages (e.g. a crash mid-compaction, or a trim that
200    // dropped the assistant tool_use block but left its tool_result).
201    // Without this the next API call fails with 400 "unexpected tool_use_id
202    // found in tool_result blocks" and the session stays bricked until the
203    // file is deleted. See #5813.
204    remove_orphaned_tool_messages(&mut state.history);
205
206    Ok(state.history)
207}

209pub fn save_interactive_session_history(path: &Path, history: &[ChatMessage]) -> Result<()> {
210    if let Some(parent) = path.parent() {
211        std::fs::create_dir_all(parent)?;
212    }
213
214    let payload = serde_json::to_string_pretty(&InteractiveSessionState::from_history(history))?;
215    std::fs::write(path, payload)?;
216    Ok(())
