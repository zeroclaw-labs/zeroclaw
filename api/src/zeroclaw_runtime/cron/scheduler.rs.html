25pub async fn run(config: Config, event_tx: EventBroadcast) -> Result<()> {
26    let poll_secs = config.reliability.scheduler_poll_secs.max(MIN_POLL_SECONDS);
27    let mut interval = time::interval(Duration::from_secs(poll_secs));
28    interval.set_missed_tick_behavior(time::MissedTickBehavior::Skip);
29    let security = Arc::new(SecurityPolicy::from_config(
30        &config.autonomy,
31        &config.workspace_dir,
32    ));
33
34    crate::health::mark_component_ok(SCHEDULER_COMPONENT);
35
36    // ── Declarative job sync: reconcile config-defined jobs with the DB.
37    let mut jobs_with_builtin = config.cron.jobs.clone();
38    if let Some(ref schedule_cron) = config.backup.schedule_cron {
39        let backup_job = CronJobDecl {
40            id: "__builtin_backup".to_string(),
41            name: Some("Scheduled backup".to_string()),
42            job_type: "shell".to_string(),
43            schedule: CronScheduleDecl::Cron {
44                expr: schedule_cron.clone(),
45                tz: config.backup.schedule_timezone.clone(),
46            },
47            command: Some("backup create".to_string()),
48            prompt: None,
49            enabled: true,
50            model: None,
51            allowed_tools: None,
52            uses_memory: true,
53            session_target: None,
54            delivery: None,
55        };
56        tracing::debug!(
57            schedule = %schedule_cron,
58            "Synthesizing builtin backup cron job from config.backup.schedule_cron"
59        );
60        jobs_with_builtin.push(backup_job);
61    }
62
63    match sync_declarative_jobs(&config, &jobs_with_builtin) {
64        Ok(()) => {
65            if !jobs_with_builtin.is_empty() {
66                tracing::info!(
67                    count = jobs_with_builtin.len(),
68                    "Synced declarative cron jobs from config"
69                );
70            }
71        }
72        Err(e) => tracing::warn!("Failed to sync declarative cron jobs: {e}"),
73    }
74
75    // ── Startup catch-up: run ALL overdue jobs before entering the
76    //    normal polling loop. The regular loop is capped by `max_tasks`,
77    //    which could leave some overdue jobs waiting across many cycles
78    //    if the machine was off for a while. The catch-up phase fetches
79    //    without the `max_tasks` limit so every missed job fires once.
80    //    Controlled by `[cron] catch_up_on_startup` (default: true).
81    if config.cron.catch_up_on_startup {
82        catch_up_overdue_jobs(&config, &security, &event_tx).await;
83    } else {
84        tracing::info!("Scheduler startup: catch-up disabled by config");
85    }
86
87    loop {
88        interval.tick().await;
89        // Keep scheduler liveness fresh even when there are no due jobs.
90        crate::health::mark_component_ok(SCHEDULER_COMPONENT);
91
92        let jobs = match due_jobs(&config, Utc::now()) {
93            Ok(jobs) => jobs,
94            Err(e) => {
95                crate::health::mark_component_error(SCHEDULER_COMPONENT, e.to_string());
96                tracing::warn!("Scheduler query failed: {e}");
97                continue;
98            }
99        };
100
101        process_due_jobs(&config, &security, jobs, SCHEDULER_COMPONENT, &event_tx).await;
102    }
103}

105/// Fetch **all** overdue jobs (ignoring `max_tasks`) and execute them.
106///
107/// Called once at scheduler startup so that jobs missed during downtime
108/// (e.g. late boot, daemon restart) are caught up immediately.
109async fn catch_up_overdue_jobs(
110    config: &Config,
111    security: &Arc<SecurityPolicy>,
112    event_tx: &EventBroadcast,
113) {
114    let now = Utc::now();
115    let jobs = match all_overdue_jobs(config, now) {
116        Ok(jobs) => jobs,
117        Err(e) => {
118            tracing::warn!("Startup catch-up query failed: {e}");
119            return;
120        }
121    };
122
123    if jobs.is_empty() {
124        tracing::info!("Scheduler startup: no overdue jobs to catch up");
125        return;
126    }
127
128    tracing::info!(
129        count = jobs.len(),
130        "Scheduler startup: catching up overdue jobs"
131    );
132
133    process_due_jobs(config, security, jobs, SCHEDULER_COMPONENT, event_tx).await;
134
135    tracing::info!("Scheduler startup: catch-up complete");
136}

143async fn execute_job_with_retry(
144    config: &Config,
145    security: &SecurityPolicy,
146    job: &CronJob,
147) -> (bool, String) {
148    let mut last_output = String::new();
149    let retries = config.reliability.scheduler_retries;
150    let mut backoff_ms = config.reliability.provider_backoff_ms.max(200);
151
152    for attempt in 0..=retries {
153        let (success, output) = match job.job_type {
154            JobType::Shell => run_job_command(config, security, job).await,
155            JobType::Agent => Box::pin(run_agent_job(config, security, job)).await,
156        };
157        last_output = output;
158
159        if success {
160            return (true, last_output);
161        }
162
163        if last_output.starts_with("blocked by security policy:") {
164            // Deterministic policy violations are not retryable.
165            return (false, last_output);
166        }
167
168        if attempt < retries {
169            let jitter_ms = u64::from(Utc::now().timestamp_subsec_millis() % 250);
170            time::sleep(Duration::from_millis(backoff_ms + jitter_ms)).await;
171            backoff_ms = (backoff_ms.saturating_mul(2)).min(30_000);
172        }
173    }
174
175    (false, last_output)
176}

178async fn process_due_jobs(
179    config: &Config,
180    security: &Arc<SecurityPolicy>,
181    jobs: Vec<CronJob>,
182    component: &str,
183    event_tx: &EventBroadcast,
184) {
185    // Refresh scheduler health on every successful poll cycle, including idle cycles.
186    crate::health::mark_component_ok(component);
187
188    let max_concurrent = config.scheduler.max_concurrent.max(1);
189    let mut in_flight = stream::iter(jobs.into_iter().map(|job| {
190        let config = config.clone();
191        let security = Arc::clone(security);
192        let component = component.to_owned();
193        async move {
194            Box::pin(execute_and_persist_job(
195                &config,
196                security.as_ref(),
197                &job,
198                &component,
199            ))
200            .await
201        }
202    }))
203    .buffer_unordered(max_concurrent);
204
205    while let Some((job_id, success, output)) = in_flight.next().await {
206        if !success {
207            tracing::warn!("Scheduler job '{job_id}' failed: {output}");
208        }
209        // Broadcast cron result to dashboard/SSE clients.
210        if let Some(tx) = event_tx {
211            let _ = tx.send(serde_json::json!({
212                "type": "cron_result",
213                "job_id": job_id,
214                "success": success,
215                "output": output,
216                "timestamp": chrono::Utc::now().to_rfc3339(),
217            }));
218        }
219    }
220}

222async fn execute_and_persist_job(
223    config: &Config,
224    security: &SecurityPolicy,
225    job: &CronJob,
226    component: &str,
227) -> (String, bool, String) {
228    crate::health::mark_component_ok(component);
229    warn_if_high_frequency_agent_job(job);
230
231    let started_at = Utc::now();
232    let (success, output) = Box::pin(execute_job_with_retry(config, security, job)).await;
233    let finished_at = Utc::now();
234    let success = Box::pin(persist_job_result(
235        config,
236        job,
237        success,
238        &output,
239        started_at,
240        finished_at,
241    ))
242    .await;
243
244    (job.id.clone(), success, output)
245}

247async fn run_agent_job(
248    config: &Config,
249    security: &SecurityPolicy,
250    job: &CronJob,
251) -> (bool, String) {
252    if !security.can_act() {
253        return (
254            false,
255            "blocked by security policy: autonomy is read-only".to_string(),
256        );
257    }
258
259    if security.is_rate_limited() {
260        return (
261            false,
262            "blocked by security policy: rate limit exceeded".to_string(),
263        );
264    }
265
266    if !security.record_action() {
267        return (
268            false,
269            "blocked by security policy: action budget exhausted".to_string(),
270        );
271    }
272    let name = job.name.clone().unwrap_or_else(|| "cron-job".to_string());
273    let prompt = job.prompt.clone().unwrap_or_default();
274
275    // Recall relevant memories so cron jobs have context awareness.
276    // Skipped when `job.uses_memory` is false (e.g. stateless digest jobs).
277    // Exclude `Conversation` memories to prevent chat context from
278    // leaking into scheduled executions (see #5415).
279    let memory_context = if !job.uses_memory {
280        String::new()
281    } else {
282        match zeroclaw_memory::create_memory(
283            &config.memory,
284            &config.workspace_dir,
285            config
286                .providers
287                .fallback_provider()
288                .and_then(|e| e.api_key.as_deref()),
289        ) {
290            Ok(mem) => match mem.recall(&prompt, 5, None, None, None).await {
291                Ok(entries) if !entries.is_empty() => {
292                    let ctx: String = entries
293                        .iter()
294                        .filter(|e| {
295                            !matches!(
296                                e.category,
297                                zeroclaw_memory::traits::MemoryCategory::Conversation
298                            )
299                        })
300                        .map(|e| format!("- {}: {}", e.key, e.content))
301                        .collect::<Vec<_>>()
302                        .join("\n");
303                    if ctx.is_empty() {
304                        String::new()
305                    } else {
306                        format!("[Memory context]\n{ctx}\n[/Memory context]\n\n")
307                    }
308                }
309                _ => String::new(),
310            },
311            Err(_) => String::new(),
312        }
313    };
314
315    let prefixed_prompt = format!("{memory_context}[cron:{} {name}] {prompt}", job.id);
316    let model_override = job.model.clone();
317
318    let mut cron_config = config.clone();
319    cron_config.memory.auto_save = false;
320
321    // Assign a unique session ID so memories written during this run can be
322    // purged atomically if the run fails (prevents snowball accumulation).
323    let run_session_id = uuid::Uuid::new_v4().to_string();
324    let session_path = std::path::PathBuf::from(format!("cron-{run_session_id}"));
325
326    let run_result = match job.session_target {
327        SessionTarget::Main | SessionTarget::Isolated => {
328            Box::pin(crate::agent::run(
329                cron_config,
330                Some(prefixed_prompt),
331                None,
332                model_override,
333                config
334                    .providers
335                    .fallback_provider()
336                    .and_then(|e| e.temperature)
337                    .unwrap_or(0.7),
338                vec![],
339                false,
340                Some(session_path.clone()),
341                job.allowed_tools.clone(),
342            ))
343            .await
344        }
345    };
346
347    match run_result {
348        Ok(response) => (
349            true,
350            if response.trim().is_empty() {
351                "agent job executed".to_string()
352            } else {
353                response
354            },
355        ),
356        Err(e) => {
357            // Purge memories written during this failed run so they don't
358            // pollute future recall and cause context snowball.
359            let mem_session_key = format!("cli:{}", session_path.display());
360            if let Ok(mem) = zeroclaw_memory::create_memory(
361                &config.memory,
362                &config.workspace_dir,
363                config
364                    .providers
365                    .fallback_provider()
366                    .and_then(|e| e.api_key.as_deref()),
367            ) {
368                let _ = mem.purge_session(&mem_session_key).await;
369            }
370            (false, format!("agent job failed: {e}"))
371        }
372    }
373}

375async fn persist_job_result(
376    config: &Config,
377    job: &CronJob,
378    mut success: bool,
379    output: &str,
380    started_at: DateTime<Utc>,
381    finished_at: DateTime<Utc>,
382) -> bool {
383    let duration_ms = (finished_at - started_at).num_milliseconds();
384
385    if let Err(e) = deliver_if_configured(config, job, output).await {
386        if job.delivery.best_effort {
387            tracing::warn!("Cron delivery failed (best_effort): {e}");
388        } else {
389            success = false;
390            tracing::warn!("Cron delivery failed: {e}");
391        }
392    }
393
394    let _ = record_run(
395        config,
396        &job.id,
397        started_at,
398        finished_at,
399        if success { "ok" } else { "error" },
400        Some(output),
401        duration_ms,
402    );
403
404    if is_one_shot_auto_delete(job) {
405        if success {
406            if let Err(e) = remove_job(config, &job.id) {
407                tracing::warn!("Failed to remove one-shot cron job after success: {e}");
408                // Fall back to disabling the job so it won't re-trigger.
409                let _ = update_job(
410                    config,
411                    &job.id,
412                    CronJobPatch {
413                        enabled: Some(false),
414                        ..CronJobPatch::default()
415                    },
416                );
417            }
418        } else {
419            let _ = record_last_run(config, &job.id, finished_at, false, output);
420            if let Err(e) = update_job(
421                config,
422                &job.id,
423                CronJobPatch {
424                    enabled: Some(false),
425                    ..CronJobPatch::default()
426                },
427            ) {
428                tracing::warn!("Failed to disable failed one-shot cron job: {e}");
429            }
430        }
431        return success;
432    }
433
434    if let Err(e) = reschedule_after_run(config, job, success, output) {
435        tracing::warn!("Failed to persist scheduler run result: {e}");
436    }
437
438    success
439}

441fn is_one_shot_auto_delete(job: &CronJob) -> bool {
442    job.delete_after_run && matches!(job.schedule, Schedule::At { .. })
443}

445fn is_high_frequency_agent_job(job: &CronJob) -> bool {
446    if !matches!(job.job_type, JobType::Agent) {
447        return false;
448    }
449    match &job.schedule {
450        Schedule::Every { every_ms } => *every_ms < 5 * 60 * 1000,
451        Schedule::Cron { .. } => {
452            let now = Utc::now();
453            next_run_for_schedule(&job.schedule, now)
454                .and_then(|a| next_run_for_schedule(&job.schedule, a).map(|b| (a, b)))
455                .map(|(a, b)| (b - a).num_minutes() < 5)
456                .unwrap_or(false)
457        }
458        Schedule::At { .. } => false,
459    }
460}

462fn warn_if_high_frequency_agent_job(job: &CronJob) {
463    if is_high_frequency_agent_job(job) {
464        tracing::warn!(
465            "Cron agent job '{}' is scheduled more frequently than every 5 minutes",
466            job.id
467        );
468    }

471async fn deliver_if_configured(config: &Config, job: &CronJob, output: &str) -> Result<()> {
472    let delivery: &DeliveryConfig = &job.delivery;
473    if !delivery.mode.eq_ignore_ascii_case("announce") {
474        return Ok(());
475    }
476
477    let channel = delivery
478        .channel
479        .as_deref()
480        .ok_or_else(|| anyhow::anyhow!("delivery.channel is required for announce mode"))?;
481    let target = delivery
482        .to
483        .as_deref()
484        .ok_or_else(|| anyhow::anyhow!("delivery.to is required for announce mode"))?;
485
486    deliver_announcement(config, channel, target, output).await
487}

489/// Delivery function type — takes owned values so the returned future is 'static.
490pub type DeliveryFn = Box<
491    dyn Fn(
492            Config,
493            String,
494            String,
495            String,
496        ) -> std::pin::Pin<Box<dyn std::future::Future<Output = Result<()>> + Send>>
497        + Send
498        + Sync,
499>;
500
501/// Global delivery function, injected by the binary crate at startup.
502static DELIVERY_FN: std::sync::OnceLock<DeliveryFn> = std::sync::OnceLock::new();
503
504/// Register the channel delivery function. Called once at startup by the binary.
505pub fn register_delivery_fn(f: DeliveryFn) {
506    let _ = DELIVERY_FN.set(f);
507}
508
509pub async fn deliver_announcement(
510    config: &Config,
511    channel: &str,
512    target: &str,
513    output: &str,
514) -> Result<()> {
515    if let Some(f) = DELIVERY_FN.get() {
516        f(
517            config.clone(),
518            channel.to_string(),
519            target.to_string(),
520            output.to_string(),
521        )
522        .await
523    } else {
524        // No handler registered: this is a runtime-level state (the binary
525        // hasn't called `register_delivery_fn`), not a per-job failure.
526        // Returning `Err` here would force every announce-mode job to set
527        // `best_effort=true` just to survive a system that legitimately has
528        // no delivery wired (e.g. headless test runs, gateway-only deployments
529        // where channel orchestration lives elsewhere).
530        //
531        // We log loudly via `tracing::warn` so operators see the dropped
532        // delivery in their logs, then return `Ok(())` so `persist_job_result`
533        // records the job execution itself as successful. Operators that
534        // actively rely on delivery wire a handler at startup; absence is a
535        // configuration signal, not a delivery error.
536        tracing::warn!(
537            channel = %channel,
538            target = %target,
539            "Cron delivery skipped: no delivery handler registered \
540             (register_delivery_fn was not called by the binary)"
541        );
542        Ok(())
543    }
544}

560async fn run_job_command_with_timeout(
561    config: &Config,
562    security: &SecurityPolicy,
563    job: &CronJob,
564    timeout: Duration,
565) -> (bool, String) {
566    if !security.can_act() {
567        return (
568            false,
569            "blocked by security policy: autonomy is read-only".to_string(),
570        );
571    }
572
573    if security.is_rate_limited() {
574        return (
575            false,
576            "blocked by security policy: rate limit exceeded".to_string(),
577        );
578    }
579
580    // Unified command validation: allowlist + risk + path checks in one call.
581    // Jobs created via the validated helpers were already checked at creation
582    // time, but we re-validate at execution time to catch policy changes and
583    // manually-edited job stores.
584    let approved = false; // scheduler runs are never pre-approved
585    if let Err(error) =
586        crate::cron::validate_shell_command_with_security(security, &job.command, approved)
587    {
588        return (false, error.to_string());
589    }
590
591    if let Some(path) = security.forbidden_path_argument(&job.command) {
592        return (
593            false,
594            format!("blocked by security policy: forbidden path argument: {path}"),
595        );
596    }
597
598    if !security.record_action() {
599        return (
600            false,
601            "blocked by security policy: action budget exhausted".to_string(),
602        );
603    }
604
605    let child = match build_cron_shell_command(&job.command, &config.workspace_dir) {
606        Ok(mut cmd) => match cmd.spawn() {
607            Ok(child) => child,
608            Err(e) => return (false, format!("spawn error: {e}")),
609        },
610        Err(e) => return (false, format!("shell setup error: {e}")),
611    };
612
613    match time::timeout(timeout, child.wait_with_output()).await {
614        Ok(Ok(output)) => {
615            let stdout = String::from_utf8_lossy(&output.stdout);
616            let stderr = String::from_utf8_lossy(&output.stderr);
617            let combined = format!(
618                "status={}\nstdout:\n{}\nstderr:\n{}",
619                output.status,
620                stdout.trim(),
621                stderr.trim()
622            );
623            (output.status.success(), combined)
624        }
625        Ok(Err(e)) => (false, format!("spawn error: {e}")),
626        Err(_) => (
627            false,
628            format!("job timed out after {}s", timeout.as_secs_f64()),
629        ),
630    }
631}

633/// Build a shell `Command` for cron job execution.
634///
635/// Uses `sh -c <command>` (non-login shell). On Windows, ZeroClaw users
636/// typically have Git Bash installed which provides `sh` in PATH, and
637/// cron commands are written with Unix shell syntax. The previous `-lc`
638/// (login shell) flag was dropped: login shells load the full user
639/// profile on every invocation which is slow and may cause side effects.
640///
641/// The command is configured with:
642/// - `current_dir` set to the workspace
643/// - `stdin` piped to `/dev/null` (no interactive input)
644/// - `stdout` and `stderr` piped for capture
645/// - `kill_on_drop(true)` for safe timeout handling
646fn build_cron_shell_command(
647    command: &str,
648    workspace_dir: &std::path::Path,
649) -> anyhow::Result<Command> {
650    let mut cmd = Command::new("sh");
651    cmd.arg("-c")
652        .arg(command)
653        .current_dir(workspace_dir)
654        .stdin(Stdio::null())
655        .stdout(Stdio::piped())
656        .stderr(Stdio::piped())
657        .kill_on_drop(true);
658
659    Ok(cmd)
660}
