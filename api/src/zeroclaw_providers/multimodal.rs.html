55/// Returns true for payloads that are plausibly loadable image references:
56/// absolute filesystem paths, `http(s)://` URLs, or base64 `data:` URIs.
57/// Placeholder-style payloads like `...`, `<path>`, or `example.png` fail
58/// this check and are left as literal text by [`parse_image_markers`], so
59/// illustrative markdown in a conversation does not trigger loader errors.
60fn is_loadable_image_reference(candidate: &str) -> bool {
61    candidate.starts_with('/')
62        || candidate.starts_with("http://")
63        || candidate.starts_with("https://")
64        || candidate.starts_with("data:")
65}

67/// Normalize a marker payload that may have been line-wrapped when pasted
68/// from a terminal (e.g. a log line where a long path was broken across
69/// rows with leading indentation). Interior newlines — and any whitespace
70/// immediately following them — are dropped; leading/trailing whitespace
71/// is trimmed. Legitimate paths may contain spaces but never newlines, so
72/// this only recovers corrupted markers and does not mangle real paths.
73fn collapse_wrapped_marker(raw: &str) -> String {
74    if !raw.contains('\n') && !raw.contains('\r') {
75        return raw.trim().to_string();
76    }
77    let mut out = String::with_capacity(raw.len());
78    let mut skip_ws = false;
79    for ch in raw.chars() {
80        if ch == '\n' || ch == '\r' {
81            skip_ws = true;
82            continue;
83        }
84        if skip_ws {
85            if ch.is_whitespace() {
86                continue;
87            }
88            skip_ws = false;
89        }
90        out.push(ch);
91    }
92    out.trim().to_string()
93}

95pub fn parse_image_markers(content: &str) -> (String, Vec<String>) {
96    let mut refs = Vec::new();
97    let mut cleaned = String::with_capacity(content.len());
98    let mut cursor = 0usize;
99
100    while let Some(rel_start) = content[cursor..].find(IMAGE_MARKER_PREFIX) {
101        let start = cursor + rel_start;
102        cleaned.push_str(&content[cursor..start]);
103
104        let marker_start = start + IMAGE_MARKER_PREFIX.len();
105        let Some(rel_end) = content[marker_start..].find(']') else {
106            cleaned.push_str(&content[start..]);
107            cursor = content.len();
108            break;
109        };
110
111        let end = marker_start + rel_end;
112        let candidate = collapse_wrapped_marker(&content[marker_start..end]);
113
114        if candidate.is_empty() || !is_loadable_image_reference(&candidate) {
115            // Preserve the original marker text (placeholders like
116            // `[IMAGE:...]` or `[IMAGE:<path>]` should survive as prose
117            // rather than triggering a loader error).
118            cleaned.push_str(&content[start..=end]);
119        } else {
120            refs.push(candidate);
121        }
122
123        cursor = end + 1;
124    }
125
126    if cursor < content.len() {
127        cleaned.push_str(&content[cursor..]);
128    }
129
130    (cleaned.trim().to_string(), refs)
131}

133pub fn count_image_markers(messages: &[ChatMessage]) -> usize {
134    messages
135        .iter()
136        .filter(|m| m.role == "user")
137        .map(|m| parse_image_markers(&m.content).1.len())
138        .sum()
139}

145pub fn extract_ollama_image_payload(image_ref: &str) -> Option<String> {
146    if image_ref.starts_with("data:") {
147        let comma_idx = image_ref.find(',')?;
148        let (_, payload) = image_ref.split_at(comma_idx + 1);
149        let payload = payload.trim();
150        if payload.is_empty() {
151            None
152        } else {
153            Some(payload.to_string())
154        }
155    } else {
156        Some(image_ref.trim().to_string()).filter(|value| !value.is_empty())
157    }
158}

160pub async fn prepare_messages_for_provider(
161    messages: &[ChatMessage],
162    config: &MultimodalConfig,
163) -> anyhow::Result<PreparedMessages> {
164    let (max_images, max_image_size_mb) = config.effective_limits();
165    let max_bytes = max_image_size_mb.saturating_mul(1024 * 1024);
166
167    let total_images = count_image_markers(messages);
168
169    if total_images == 0 {
170        return Ok(PreparedMessages {
171            messages: messages.to_vec(),
172            contains_images: false,
173        });
174    }
175
176    // When image count exceeds the limit, strip markers from oldest messages
177    // first so that the most recent (most relevant) images survive. This
178    // prevents conversations from becoming permanently stuck once the
179    // cumulative image count crosses the threshold.
180    let trimmed = if total_images > max_images {
181        trim_old_images(messages, max_images)
182    } else {
183        messages.to_vec()
184    };
185
186    let remote_client = build_runtime_proxy_client_with_timeouts("provider.ollama", 30, 10);
187
188    let mut normalized_messages = Vec::with_capacity(trimmed.len());
189    for message in &trimmed {
190        if message.role != "user" {
191            normalized_messages.push(message.clone());
192            continue;
193        }
194
195        let (cleaned_text, refs) = parse_image_markers(&message.content);
196        if refs.is_empty() {
197            normalized_messages.push(message.clone());
198            continue;
199        }
200
201        let mut normalized_refs = Vec::with_capacity(refs.len());
202        for reference in refs {
203            let data_uri =
204                normalize_image_reference(&reference, config, max_bytes, &remote_client).await?;
205            normalized_refs.push(data_uri);
206        }
207
208        let content = compose_multimodal_message(&cleaned_text, &normalized_refs);
209        normalized_messages.push(ChatMessage {
210            role: message.role.clone(),
211            content,
212        });
213    }
214
215    Ok(PreparedMessages {
216        messages: normalized_messages,
217        contains_images: true,
218    })
219}

221/// Strip image markers from older messages (oldest first) until total image
222/// count is within `max_images`. Keeps the text content of each message.
223fn trim_old_images(messages: &[ChatMessage], max_images: usize) -> Vec<ChatMessage> {
224    // Find which messages (by index) contain images, oldest first.
225    let image_positions: Vec<(usize, usize)> = messages
226        .iter()
227        .enumerate()
228        .filter(|(_, m)| m.role == "user")
229        .filter_map(|(i, m)| {
230            let count = parse_image_markers(&m.content).1.len();
231            if count > 0 { Some((i, count)) } else { None }
232        })
233        .collect();
234
235    // Determine how many images to drop (from the oldest messages).
236    let total: usize = image_positions.iter().map(|(_, c)| c).sum();
237    let mut to_drop = total.saturating_sub(max_images);
238
239    // Collect indices of messages whose images should be stripped.
240    let mut strip_indices = std::collections::HashSet::new();
241    for &(idx, count) in &image_positions {
242        if to_drop == 0 {
243            break;
244        }
245        strip_indices.insert(idx);
246        to_drop = to_drop.saturating_sub(count);
247    }
248
249    messages
250        .iter()
251        .enumerate()
252        .map(|(i, m)| {
253            if strip_indices.contains(&i) {
254                let (cleaned, _) = parse_image_markers(&m.content);
255                let text = if cleaned.trim().is_empty() {
256                    "[image removed from history]".to_string()
257                } else {
258                    cleaned
259                };
260                ChatMessage {
261                    role: m.role.clone(),
262                    content: text,
263                }
264            } else {
265                m.clone()
266            }
267        })
268        .collect()
269}

271fn compose_multimodal_message(text: &str, data_uris: &[String]) -> String {
272    let mut content = String::new();
273    let trimmed = text.trim();
274
275    if !trimmed.is_empty() {
276        content.push_str(trimmed);
277        content.push_str("\n\n");
278    }
279
280    for (index, data_uri) in data_uris.iter().enumerate() {
281        if index > 0 {
282            content.push('\n');
283        }
284        content.push_str(IMAGE_MARKER_PREFIX);
285        content.push_str(data_uri);
286        content.push(']');
287    }
288
289    content
290}

292async fn normalize_image_reference(
293    source: &str,
294    config: &MultimodalConfig,
295    max_bytes: usize,
296    remote_client: &Client,
297) -> anyhow::Result<String> {
298    if source.starts_with("data:") {
299        return normalize_data_uri(source, max_bytes);
300    }
301
302    if source.starts_with("http://") || source.starts_with("https://") {
303        if !config.allow_remote_fetch {
304            return Err(MultimodalError::RemoteFetchDisabled {
305                input: source.to_string(),
306            }
307            .into());
308        }
309
310        return normalize_remote_image(source, max_bytes, remote_client).await;
311    }
312
313    normalize_local_image(source, max_bytes).await
314}

316fn normalize_data_uri(source: &str, max_bytes: usize) -> anyhow::Result<String> {
317    let Some(comma_idx) = source.find(',') else {
318        return Err(MultimodalError::InvalidMarker {
319            input: source.to_string(),
320            reason: "expected data URI payload".to_string(),
321        }
322        .into());
323    };
324
325    let header = &source[..comma_idx];
326    let payload = source[comma_idx + 1..].trim();
327
328    if !header.contains(";base64") {
329        return Err(MultimodalError::InvalidMarker {
330            input: source.to_string(),
331            reason: "only base64 data URIs are supported".to_string(),
332        }
333        .into());
334    }
335
336    let mime = header
337        .trim_start_matches("data:")
338        .split(';')
339        .next()
340        .unwrap_or_default()
341        .trim()
342        .to_ascii_lowercase();
343
344    validate_mime(source, &mime)?;
345
346    let decoded = STANDARD
347        .decode(payload)
348        .map_err(|error| MultimodalError::InvalidMarker {
349            input: source.to_string(),
350            reason: format!("invalid base64 payload: {error}"),
351        })?;
352
353    validate_size(source, decoded.len(), max_bytes)?;
354
355    Ok(format!("data:{mime};base64,{}", STANDARD.encode(decoded)))
356}

358async fn normalize_remote_image(
359    source: &str,
360    max_bytes: usize,
361    remote_client: &Client,
362) -> anyhow::Result<String> {
363    let response = remote_client.get(source).send().await.map_err(|error| {
364        MultimodalError::RemoteFetchFailed {
365            input: source.to_string(),
366            reason: error.to_string(),
367        }
368    })?;
369
370    let status = response.status();
371    if !status.is_success() {
372        return Err(MultimodalError::RemoteFetchFailed {
373            input: source.to_string(),
374            reason: format!("HTTP {status}"),
375        }
376        .into());
377    }
378
379    if let Some(content_length) = response.content_length() {
380        let content_length = usize::try_from(content_length).unwrap_or(usize::MAX);
381        validate_size(source, content_length, max_bytes)?;
382    }
383
384    let content_type = response
385        .headers()
386        .get(reqwest::header::CONTENT_TYPE)
387        .and_then(|value| value.to_str().ok())
388        .map(ToString::to_string);
389
390    let bytes = response
391        .bytes()
392        .await
393        .map_err(|error| MultimodalError::RemoteFetchFailed {
394            input: source.to_string(),
395            reason: error.to_string(),
396        })?;
397
398    validate_size(source, bytes.len(), max_bytes)?;
399
400    let mime = detect_mime(None, bytes.as_ref(), content_type.as_deref()).ok_or_else(|| {
401        MultimodalError::UnsupportedMime {
402            input: source.to_string(),
403            mime: "unknown".to_string(),
404        }
405    })?;
406
407    validate_mime(source, &mime)?;
408
409    Ok(format!("data:{mime};base64,{}", STANDARD.encode(bytes)))
410}

412async fn normalize_local_image(source: &str, max_bytes: usize) -> anyhow::Result<String> {
413    let path = Path::new(source);
414    if !path.exists() || !path.is_file() {
415        return Err(MultimodalError::ImageSourceNotFound {
416            input: source.to_string(),
417        }
418        .into());
419    }
420
421    let metadata =
422        tokio::fs::metadata(path)
423            .await
424            .map_err(|error| MultimodalError::LocalReadFailed {
425                input: source.to_string(),
426                reason: error.to_string(),
427            })?;
428
429    validate_size(
430        source,
431        usize::try_from(metadata.len()).unwrap_or(usize::MAX),
432        max_bytes,
433    )?;
434
435    let bytes = tokio::fs::read(path)
436        .await
437        .map_err(|error| MultimodalError::LocalReadFailed {
438            input: source.to_string(),
439            reason: error.to_string(),
440        })?;
441
442    validate_size(source, bytes.len(), max_bytes)?;
443
444    let mime =
445        detect_mime(Some(path), &bytes, None).ok_or_else(|| MultimodalError::UnsupportedMime {
446            input: source.to_string(),
447            mime: "unknown".to_string(),
448        })?;
449
450    validate_mime(source, &mime)?;
451
452    Ok(format!("data:{mime};base64,{}", STANDARD.encode(bytes)))
453}

455fn validate_size(source: &str, size_bytes: usize, max_bytes: usize) -> anyhow::Result<()> {
456    if size_bytes > max_bytes {
457        return Err(MultimodalError::ImageTooLarge {
458            input: source.to_string(),
459            size_bytes,
460            max_bytes,
461        }
462        .into());
463    }
464
465    Ok(())
466}

468fn validate_mime(source: &str, mime: &str) -> anyhow::Result<()> {
469    if ALLOWED_IMAGE_MIME_TYPES.contains(&mime) {
470        return Ok(());
471    }
472
473    Err(MultimodalError::UnsupportedMime {
474        input: source.to_string(),
475        mime: mime.to_string(),
476    }
477    .into())
478}

480fn detect_mime(
481    path: Option<&Path>,
482    bytes: &[u8],
483    header_content_type: Option<&str>,
484) -> Option<String> {
485    if let Some(header_mime) = header_content_type.and_then(normalize_content_type) {
486        return Some(header_mime);
487    }
488
489    if let Some(path) = path
490        && let Some(ext) = path.extension().and_then(|value| value.to_str())
491        && let Some(mime) = mime_from_extension(ext)
492    {
493        return Some(mime.to_string());
494    }
495
496    mime_from_magic(bytes).map(ToString::to_string)
497}
498
499fn normalize_content_type(content_type: &str) -> Option<String> {
500    let mime = content_type.split(';').next()?.trim().to_ascii_lowercase();
501    if mime.is_empty() { None } else { Some(mime) }
502}
503
504fn mime_from_extension(ext: &str) -> Option<&'static str> {
505    match ext.to_ascii_lowercase().as_str() {
506        "png" => Some("image/png"),
507        "jpg" | "jpeg" => Some("image/jpeg"),
508        "webp" => Some("image/webp"),
509        "gif" => Some("image/gif"),
510        "bmp" => Some("image/bmp"),
511        _ => None,
512    }
513}
514
515fn mime_from_magic(bytes: &[u8]) -> Option<&'static str> {
516    if bytes.len() >= 8 && bytes.starts_with(&[0x89, b'P', b'N', b'G', b'\r', b'\n', 0x1a, b'\n']) {
517        return Some("image/png");
518    }
519
520    if bytes.len() >= 3 && bytes.starts_with(&[0xff, 0xd8, 0xff]) {
521        return Some("image/jpeg");
522    }
523
524    if bytes.len() >= 6 && (bytes.starts_with(b"GIF87a") || bytes.starts_with(b"GIF89a")) {
525        return Some("image/gif");
526    }
527
528    if bytes.len() >= 12 && bytes.starts_with(b"RIFF") && &bytes[8..12] == b"WEBP" {
529        return Some("image/webp");
530    }
531
532    if bytes.len() >= 2 && bytes.starts_with(b"BM") {
533        return Some("image/bmp");
534    }
535
536    None
537}

807    #[tokio::test]
808    async fn prepare_messages_trims_then_normalizes_surviving_images() {
809        // End-to-end: 3 images, max 2. After trimming the oldest, the two
810        // surviving images should be normalized (base64-encoded) successfully.
811        let temp = tempfile::tempdir().unwrap();
812        let mut paths = Vec::new();
813        for name in ["old.png", "mid.png", "new.png"] {
814            let p = temp.path().join(name);
815            // Minimal valid PNG (1x1 white pixel)
816            let png_data = [
817                0x89, 0x50, 0x4E, 0x47, 0x0D, 0x0A, 0x1A, 0x0A, // PNG signature
818                0x00, 0x00, 0x00, 0x0D, 0x49, 0x48, 0x44, 0x52, // IHDR chunk
819                0x00, 0x00, 0x00, 0x01, 0x00, 0x00, 0x00, 0x01, 0x08, 0x02, 0x00, 0x00, 0x00, 0x90,
820                0x77, 0x53, 0xDE, // 1x1 RGB
821                0x00, 0x00, 0x00, 0x0C, 0x49, 0x44, 0x41, 0x54, // IDAT chunk
822                0x08, 0xD7, 0x63, 0xF8, 0xCF, 0xC0, 0x00, 0x00, 0x00, 0x02, 0x00, 0x01, 0xE2, 0x21,
823                0xBC, 0x33, // IDAT data + CRC
824                0x00, 0x00, 0x00, 0x00, 0x49, 0x45, 0x4E, 0x44, // IEND chunk
825                0xAE, 0x42, 0x60, 0x82,
826            ];
827            std::fs::write(&p, png_data).unwrap();
828            paths.push(p);
829        }
830
831        let messages = vec![
832            ChatMessage::user(format!("[IMAGE:{}]\nOld", paths[0].display())),
833            ChatMessage::user(format!("[IMAGE:{}]\nMid", paths[1].display())),
834            ChatMessage::user(format!("[IMAGE:{}]\nNew", paths[2].display())),
835        ];
836
837        let config = MultimodalConfig {
838            max_images: 2,
839            max_image_size_mb: 5,
840            allow_remote_fetch: false,
841            ..Default::default()
842        };
843
844        let result = prepare_messages_for_provider(&messages, &config)
845            .await
846            .expect("should succeed after trimming");
847
848        assert!(result.contains_images);
849        assert_eq!(result.messages.len(), 3);
850        // First message should have image stripped, text preserved
851        assert!(!result.messages[0].content.contains("data:image"));
852        assert!(result.messages[0].content.contains("Old"));
853        // Second and third should have base64-encoded images
854        assert!(result.messages[1].content.contains("data:image"));
855        assert!(result.messages[2].content.contains("data:image"));
856    }
