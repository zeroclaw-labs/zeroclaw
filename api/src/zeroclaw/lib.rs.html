115Start the gateway server (webhooks, websockets).
116
117Runs the HTTP/WebSocket gateway that accepts incoming webhook events \
118and WebSocket connections. Bind address defaults to the values in \

120
121Examples:
122  zeroclaw gateway start              # use config defaults
123  zeroclaw gateway start -p 8080      # listen on port 8080

133        #[arg(long)]
134        host: Option<String>,
135    },
136    /// Restart the gateway server
137    #[command(long_about = "\
138Restart the gateway server.
139
140Stops the running gateway if present, then starts a new instance \
141with the current configuration.
142
143Examples:
144  zeroclaw gateway restart            # restart with config defaults
145  zeroclaw gateway restart -p 8080    # restart on port 8080")]
146    Restart {
147        /// Port to listen on (use 0 for random available port); defaults to config gateway.port
148        #[arg(short, long)]
149        port: Option<u16>,
150
151        /// Host to bind to; defaults to config gateway.host
152        /// Note: Binding to 0.0.0.0 requires `gateway.allow_public_bind = true` in config
153        #[arg(long)]
154        host: Option<String>,
155    },
156    /// Show or generate the pairing code without restarting
157    #[command(long_about = "\
158Show or generate the gateway pairing code.
159
160Displays the pairing code for connecting new clients without \
161restarting the gateway. Requires the gateway to be running.
162
163With --new, generates a fresh pairing code even if the gateway \
164was previously paired (useful for adding additional clients).
165
166Examples:
167  zeroclaw gateway get-paircode       # show current pairing code
168  zeroclaw gateway get-paircode --new # generate a new pairing code")]
169    GetPaircode {
170        /// Generate a new pairing code (even if already paired)
171        #[arg(long)]
