4pub fn run() -> anyhow::Result<()> {
5    let root = repo_root();
6    let locales_dir = fluent_locales_dir(&root);
7
8    if !locales_dir.exists() {
9        anyhow::bail!("locales dir not found: {}", locales_dir.display());
10    }
11
12    let mut error_count = 0;
13
14    for locale in fluent_locales(&root)? {
15        let locale_dir = locales_dir.join(&locale);
16        for ftl_path in ftl_files_in(&locale_dir)? {
17            let errors = check_ftl_file(&ftl_path)?;
18            if errors.is_empty() {
19                println!("ok  {}", ftl_path.display());
20            } else {
21                for (line_no, msg) in &errors {
22                    eprintln!("{}:{}: {}", ftl_path.display(), line_no, msg);
23                    error_count += 1;
24                }
25            }
26        }
27    }
28
29    if error_count > 0 {
30        anyhow::bail!("{error_count} FTL syntax error(s) found");
31    }
32    Ok(())
33}

35fn check_ftl_file(path: &Path) -> anyhow::Result<Vec<(usize, String)>> {
36    let src = std::fs::read_to_string(path)?;
37    let mut errors = vec![];
38
39    for (i, line) in src.lines().enumerate() {
40        let line_no = i + 1;
41        let trimmed = line.trim();
42
43        if trimmed.is_empty() || trimmed.starts_with('#') {
44            continue;
45        }
46
47        // Key lines must match: identifier = value
48        if let Some((key, value)) = trimmed.split_once(" = ") {
49            // Key must be alphanumeric + hyphens only
50            if !key.chars().all(|c| c.is_alphanumeric() || c == '-') {
51                errors.push((line_no, format!("invalid key '{key}'")));
52            }
53            // Check for unescaped braces in value (Fluent requires {"{"}...{"}"})
54            check_unescaped_braces(value, line_no, &mut errors);
55        } else if !trimmed.starts_with('-') {
56            errors.push((
57                line_no,
58                format!("malformed line (expected 'key = value'): {trimmed}"),
59            ));
60        }
61    }
62
63    Ok(errors)
64}

66fn check_unescaped_braces(value: &str, line_no: usize, errors: &mut Vec<(usize, String)>) {
67    let chars: Vec<char> = value.chars().collect();
68    let mut i = 0;
69    while i < chars.len() {
70        match chars[i] {
71            '{' => {
72                // Check if it's a Fluent placeable: {"{"} or {some-term} etc.
73                // If the next char is '"' it's an escaped literal — skip to closing }
74                if i + 1 < chars.len() && chars[i + 1] == '"' {
75                    // find closing "}
76                    if let Some(close) = chars[i + 2..].iter().position(|&c| c == '"') {
77                        i += 2 + close + 1; // skip past closing "
78                        // skip the closing }
79                        if i < chars.len() && chars[i] == '}' {
80                            i += 1;
81                        }
82                    } else {
83                        errors.push((line_no, "unclosed Fluent escape {\"...\"".to_string()));
84                        i += 1;
85                    }
86                } else {
87                    // Any other { is treated as an unescaped brace — Fluent placements
88                    // are valid (e.g. {$var}, {-term}) but naked { without . or - or $
89                    // followed by non-identifier chars is suspicious. For our simple
90                    // tools.ftl which uses only escaped literals, flag bare {.
91                    let peek = chars.get(i + 1).copied();
92                    if !matches!(peek, Some('$' | '-' | '.')) {
93                        errors.push((
94                            line_no,
95                            format!(
96                                "unescaped '{{' at col {i}; use {{\"{{\"}} for a literal brace"
97                            ),
98                        ));
99                    }
100                    i += 1;
101                }
102            }
103            _ => i += 1,
104        }
105    }
